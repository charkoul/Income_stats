CREATE TABLE `transactionshistory` (
  `ID` int(11) NOT NULL AUTO_INCREMENT,
  `TUN` varchar(500) NOT NULL,
  `BANK` int(2) DEFAULT NULL,
  `TRANSACTIONDATE` date NOT NULL,
  `TRANSACTIONDESCRIPTION` varchar(1000) DEFAULT NULL,
  `TRANSACTIONCOMMENT` varchar(1000) DEFAULT NULL,
  `AMOUNT` decimal(13,2) NOT NULL,
  `ACCOUNTNUMBER` varchar(100) DEFAULT NULL,
  `TRANSACTIONNUMBER` varchar(500) DEFAULT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE KEY `TUN_UNIQUE` (`TUN`),
  KEY `BANK_FK_idx` (`BANK`),
  CONSTRAINT `BANK_FK` FOREIGN KEY (`BANK`) REFERENCES `bank` (`ID`)
) ENGINE=InnoDB AUTO_INCREMENT=1235 DEFAULT CHARSET=greek
